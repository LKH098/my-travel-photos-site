        <div id="pagination" class="flex justify-center mt-8"></div>

    </main>


    <script>

        // --- 設定你的 GitHub 倉庫資訊 ---

        const owner = 'lkh098'; // 例如: 'your-username'

        const repo = 'my-travel-photos-site';       // 例如: 'my-travel-photos-site'

        const imageFolderPath = 'images';  // 圖片在倉庫中的路徑，例如 'images'


        const imagesPerPage = 9;

        let currentPage = 1;

        let allImageData = []; // 用於儲存從 GitHub API 獲取的所有圖片資料


        const imageGrid = document.getElementById('image-grid');

        const paginationDiv = document.getElementById('pagination');


        /**

         * 從 GitHub API 獲取圖片檔案列表

         */

        async function fetchImagesFromGitHub() {

            const url = `https://api.github.com/repos/<span class="math-inline">\{owner\}/</span>{repo}/contents/${imageFolderPath}`;

            try {

                const response = await fetch(url);

                if (!response.ok) {

                    throw new Error(`GitHub API 錯誤: ${response.statusText}`);

                }

                const data = await response.json();


                // 過濾出圖片檔案，並建立圖片資料物件

                allImageData = data

                    .filter(item => item.type === 'file' && /\.(png|jpg|jpeg|gif|webp|avif)$/i.test(item.name))

                    .sort((a, b) => a.name.localeCompare(b.name)) // 按檔名排序

                    .map(item => ({

                        src: item.download_url, // GitHub 提供直接下載的 URL

                        alt: item.name,

                        location: item.name.split('.')[0].replace(/[-_]/g, ' '), // 預設為檔名，去除副檔名和連字號/底線

                        link: '#' // 預設連結，需要手動或透過其他方式補充

                    }));


                // 獲取圖片後，再顯示第一頁圖片並渲染分頁

                displayImages(currentPage);

                renderPagination();


            } catch (error) {

                console.error("獲取圖片失敗:", error);

                imageGrid.innerHTML = '<p class="text-red-500 text-center">無法載入圖片。請檢查 GitHub 設定或網路連線。</p>';

            }

        }


        /**

         * 根據頁碼顯示對應的圖片

         * @param {number} page - 要顯示的頁碼

         */

        function displayImages(page) {

            imageGrid.innerHTML = '';

            const startIndex = (page - 1) * imagesPerPage;

            const endIndex = startIndex + imagesPerPage;

            const currentImages = allImageData.slice(startIndex, endIndex);


            currentImages.forEach(image => {

                const imageContainer = document.createElement('div');

                imageContainer.classList.add(

                    'image-container', 'relative', 'overflow-hidden', 'rounded-md', 'shadow-lg', 'bg-gray-800'

                );


                const img = document.createElement('img');

                img.src = image.src;

                img.alt = image.alt;

                img.classList.add('w-full', 'h-auto', 'block');


                const overlay = document.createElement('div');

                overlay.classList.add('overlay');


                const link = document.createElement('a');

                link.href = image.link;

                link.textContent = image.location;

                link.target = '_blank';

                link.rel = 'noopener noreferrer';

                overlay.appendChild(link);


                imageContainer.appendChild(img);

                imageContainer.appendChild(overlay);

                imageGrid.appendChild(imageContainer);

            });

        }


        /**

         * 渲染分頁按鈕

         */

        function renderPagination() {

            paginationDiv.innerHTML = '';

            const totalPages = Math.ceil(allImageData.length / imagesPerPage);


            for (let i = 1; i <= totalPages; i++) {

                const button = document.createElement('button');

                button.textContent = i;

                button.classList.add('px-4', 'py-2', 'mx-1', 'rounded', 'focus:outline-none', 'transition-colors', 'duration-200');


                if (i === currentPage) {

                    button.classList.add('bg-gray-700', 'text-white', 'shadow-md');

                } else {

                    button.classList.add('bg-gray-300', 'text-gray-800', 'hover:bg-gray-400');

                    button.addEventListener('click', () => {

                        currentPage = i;

                        displayImages(currentPage);

                        renderPagination();

                        window.scrollTo({ top: 0, behavior: 'smooth' });

                    });

                }

                paginationDiv.appendChild(button);

            }

        }


        // --- 頁面載入時執行 ---

        // 呼叫函式從 GitHub 獲取圖片並初始化網站

        fetchImagesFromGitHub();

    </script>
